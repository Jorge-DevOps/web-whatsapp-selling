🚀 DOCKER BUILD & RUN
════════════════════════════════════════════════════════════════════════════

Build imagen:
  docker build -t whatsapp-selling-bot:latest .
  docker build -t whatsapp-selling-bot:v1.0 .

Run contenedor:
  docker run -it whatsapp-selling-bot:latest
  
  Con volúmenes persistentes:
  docker run -d \
    --name whatsapp-bot \
    -v whatsapp_sessions:/app/.wwebjs_cache \
    -v whatsapp_auth:/app/.wwebjs_auth \
    -v bot_logs:/app/logs \
    --restart unless-stopped \
    whatsapp-selling-bot:latest

Docker Compose:
  docker-compose up -d
  docker-compose logs -f whatsapp-bot
  docker-compose down

📊 IMAGEN DOCKER
════════════════════════════════════════════════════════════════════════════

Base: node:18-alpine
Tamaño: ~200MB (optimizado)

Layers:
  1. FROM node:18-alpine
  2. WORKDIR /app
  3. COPY package*.json
  4. RUN npm ci --only=production
  5. COPY src ./src
  6. RUN mkdir -p dirs
  7. EXPOSE 8080
  8. HEALTHCHECK
  9. CMD ["node", "src/index.js"]
